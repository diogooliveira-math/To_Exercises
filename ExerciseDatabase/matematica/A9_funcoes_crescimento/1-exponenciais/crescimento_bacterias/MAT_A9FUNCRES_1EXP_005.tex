\begin{enumerate}
  \item Calcula número às 9h e às 12h.
  \vspace{3cm}
  \item Determina tempo até 15000 bactérias (horas e minutos).
  \vspace{3cm}
  \item Esboça o gráfico e interpreta três pontos.
  \vspace{3cm}
  \item Cria um problema adicional de crescimento e resolve.
  \vspace{3cm}
\end{enumerate}
